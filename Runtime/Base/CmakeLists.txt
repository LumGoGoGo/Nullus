# 收集头文件和源文件
###############################################################################
file(GLOB_RECURSE BASE_HEAD_FILES
  ${CMAKE_CURRENT_SOURCE_DIR}/*.h
  ${CMAKE_CURRENT_SOURCE_DIR}/*.hpp
  ${CMAKE_CURRENT_SOURCE_DIR}/*.inl
)
file(GLOB_RECURSE BASE_SOURCE_FILES
  ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp
  ${CMAKE_CURRENT_SOURCE_DIR}/*.c
)

# 对文件进行分组
###############################################################################
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${BASE_HEAD_FILES})
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${BASE_SOURCE_FILES})

if(NLS_USE_SHARED_LIBRARY)
add_library(NLS_Base SHARED ${BASE_HEAD_FILES} ${BASE_SOURCE_FILES})

target_compile_definitions(
  NLS_Base
  PRIVATE
  NLS_BASE_EXPORT
)
else()
add_library(NLS_Base STATIC ${BASE_HEAD_FILES} ${BASE_SOURCE_FILES})
endif()

# 设置包含路径
###############################################################################
target_include_directories(
  NLS_Base
  PUBLIC # 其他项目使用头文件时所需的目录
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    ${NLS_ThirdParty_DIR}
)

# 指定输出路径
###############################################################################
set_target_properties(NLS_Base PROPERTIES
  RUNTIME_OUTPUT_DIRECTORY "${NLS_APP_OUTPUT_PATH}"
)
